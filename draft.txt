// Код для записи
#include <Arduino.h>
#include <driver/i2s.h>
#include <SPI.h>
#include <SD.h>
#include <cmath>

// Конфигурация пинов микрофона
#define I2S_BCLK_PIN 14
#define I2S_LRC_PIN  15
#define I2S_DATA_PIN 32 // DIN для динамика или DOUT для микрофона

// Конфигурация пина светодиода
#define BLUE_LED 2

// Конфигурация пинов для SD-карты
#define SD_CS_PIN 5
#define SD_MOSI_PIN 23
#define SD_MISO_PIN 19
#define SD_SCK_PIN  18

// Название файла для записи в SD-карту и чтения из неё
const char* filename = "/audio.txt";

// Буфер для чтения данных
#define READ_BUFFER_SIZE 1024
uint8_t readBuffer[READ_BUFFER_SIZE];

// Период для таймера
const int TIMER_PERIOD = 100;

// Пороговое значения для включения
const int LIMIT_VALUE = 50;

// Последние зафиксированные значения для работы по таймеру
int prev_time = 0;
float prev_sum = 0;

// Среднее значение амплитуды за 100 мс и кол-во сэмплов
float sum = 0, count = 4608;

// Начало I2S
void initI2S() 
{
  // 1. Конфигурация I2S драйвера
  const i2s_config_t i2s_config = {
    .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
    .sample_rate = 44100,
    .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,
    .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
    .communication_format = I2S_COMM_FORMAT_STAND_I2S,
    .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1, // Уровень прерывания
    .dma_buf_count = 8,
    .dma_buf_len = 64,
    .use_apll = false,
    .tx_desc_auto_clear = true, // Очищать буфер при отсутствии данных (тишина)
    .fixed_mclk = 0
  };

  // 2. Установка драйвера
  // I2S_NUM_0 - номер порта
  esp_err_t err = i2s_driver_install(I2S_NUM_0, &i2s_config, 0, NULL);
    if (err != ESP_OK) {
    Serial.printf("Установка драйвера провалилась: %d\n", err);
    while (true);
  }

  // 3. Конфигурация пинов
  const i2s_pin_config_t pin_config = {
    .bck_io_num = I2S_BCLK_PIN,
    .ws_io_num = I2S_LRC_PIN,
    .data_out_num = I2S_PIN_NO_CHANGE,
    .data_in_num = I2S_DATA_PIN
  };
  
  i2s_set_pin(I2S_NUM_0, &pin_config);
  
  Serial.println("I2S драйвер установлен");
  delay(100); // Чтобы не было мусора в первые миллисекунды записи
}

// Остановка I2S
void stopI2S() {
  i2s_stop(I2S_NUM_0);
  i2s_driver_uninstall(I2S_NUM_0);
  delay(10);  // Маленькая пауза для стабилизации
}

// Начало SPI интерфейса для SD-карты
void initSPI()
{
  SPI.begin(SD_SCK_PIN, SD_MISO_PIN, SD_MOSI_PIN, SD_CS_PIN);
  if (!SD.begin(SD_CS_PIN, SPI, 4000000)) {
    Serial.println("Ошибка монтирования SD карты");
    while(true);
  }
  Serial.println("Монтирование SD-карты прошло успешно");
  delay(50); // У карты есть своя инерция для записи и чтения
}

void append(float value) 
{
  File file = SD.open(filename, FILE_APPEND);
  if (!file) {
    Serial.println("Ошибка файла для записи");
    while(true);
  } 
  else {
    Serial.println("Успешно записано");
    file.println(value);
    file.close();
  }
}

void setup() 
{
  
  Serial.begin(115200);
  delay(1000); // Чтобы USB успел отреагировать
  
  initSPI();
  File file = SD.open(filename, FILE_WRITE);
  if (!file) {
    Serial.println("Ошибка файла для записи");
    while(true);
  } 
  else {
    Serial.println("Запись в файл и чтение из неё возможны");
    file.close();
  }

  initI2S();
  
  pinMode(BLUE_LED, OUTPUT);
}

void loop() 
{
  size_t bytesRead = 0;
  
  // Чтение блока данных из I2S
  esp_err_t result = i2s_read(I2S_NUM_0, &readBuffer, READ_BUFFER_SIZE, &bytesRead, portMAX_DELAY);
  
  if (result == ESP_OK && bytesRead > 0) 
  {

    int samplesRead = bytesRead / sizeof(int16_t);
    int16_t* samples = (int16_t*)readBuffer;
    
    // Проходим по буферу и выводим значения (стерео, берем левый канал)
    for (int i = 0; i < samplesRead; i += 2) {
      // Вывод в Serial Plotter
      
      sum += abs(samples[i]);

      // Внимание: При высокой частоте дискретизации (>16кГц) 
      // вывод в Serial может замедлять цикл. 
      //Для теста лучше использовать меньшую частоту или выводить каждый N-й сэмпл.
      
    }

    if (millis() - prev_time >= TIMER_PERIOD) 
    {

      stopI2S();
      append(sum / count);

      if ((sum / count) / (prev_sum / count) >= LIMIT_VALUE) {
        digitalWrite(BLUE_LED, HIGH);
      }
      else {
        digitalWrite(BLUE_LED, LOW);
      }

      initI2S();
      prev_sum = sum;
      sum = 0.0;
      prev_time = millis();

    }

  }
  
}
